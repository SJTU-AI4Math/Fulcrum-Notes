        \subsection{整除}
            
            \begin{dfn}
                [Div]
                {整除}
                [Divides]
                [猫猫]
                \DEF*
                    [ \(a, b:\Z\), \(a\neq 0\)]
                    { \(a\) 整除 \(b\) }
                    {\(\exists[n][b=an][\Z]\)}
                    [ \(a\mid b\)]

                \DEF*
                    { \(a\) 不整除 \(b\) }
                    {\(\neg(a\mid b)\)}
                    [ \(a\nmid b\)]
            \end{dfn}
            
            \begin{ppt}
                [DivPO]
                {整除的偏序性}
                [Partial Order of Divisibility]
                [猫猫]
                \THM
                    {整除关系是偏序关系}
            \end{ppt}
            
            \begin{ppt}
                [DivLinear]
                {整除的线性性}
                [Linearity of Divisibility]
                [猫猫]

                    {整除关系是线性关系}
            \end{ppt}
